1. docker-compose run django django-admin startproject core . # for pull all image and setup project
2. docker-compose down # stop docker-compose
3. docker-compose up
4. docker exec -it django bash # access to django container it's the same directory we have here
5. python manage.py migrate # setup container django environment
6. it's ready
